% --------------{ Score / rate bar}--------------- %
\newcommand\score[2]{
  \pgfmathsetmacro\pgfxa{#1+1}
  \tikz{%
    \foreach \i in {1,...,#2} {
      \pgfmathparse{(\i<=#1?"black":"white")}
      \edef\scorecolor{\pgfmathresult}
      \draw[color=black,fill=\scorecolor](\i*0.6em,0) circle (0.3ex);
    }%
  }%
}
